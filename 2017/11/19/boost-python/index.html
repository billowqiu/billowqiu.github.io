struct VecConvert
{
    typedef std::vector<T> VecT;
    static PyObject* convert(const VecT& vt)
    {
        boost::python::list pylist;
        for (auto& t : vt)
        {
            pylist.append(t);
        }
        return boost::python::incref(pylist.ptr());
    }
};
